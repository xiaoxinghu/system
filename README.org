Create nix shell file =shell.nix=.

#+begin_src nix
# shell.nix
{ pkgs ? import <nixpkgs> {
  config = {
    permittedinsecurepackages = [
      "nodejs-16.20.2"
    ];
  };
}}:

pkgs.mkshell {
  buildinputs = with pkgs; [
    nodejs-16_x
  ];
  shellhook = ''
    echo "node version: $(node --version)"
  '';
}
#+end_src
